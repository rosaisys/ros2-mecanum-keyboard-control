# ROS2四大核心整合指南 - 智能车实际应用

## 概述
本指南将ROS2四大核心机制（节点/话题/服务/动作）整合到智能车实际应用中，提供完整的项目结构和实际部署建议。

## 2. 整合控制节点示例

### 功能描述
整合节点将话题发布、服务提供和动作处理集成到一个节点中，模拟智能车的主控节点。

### 代码实现 (`integrated_control_node.cpp`)

```cpp
#include "rclcpp/rclcpp.hpp"
#include "rclcpp_action/rclcpp_action.hpp"
#include "geometry_msgs/msg/twist.hpp"
#include "std_msgs/msg/float32.hpp"
#include "smart_car_core_demo/srv/emergency_stop.hpp"
#include "smart_car_core_demo/action/countdown_park.hpp"

#include <memory>
#include <thread>
#include <chrono>

using namespace std::chrono_literals;
using EmergencyStop = smart_car_core_demo::srv::EmergencyStop;
using CountdownPark = smart_car_core_demo::action::CountdownPark;
using GoalHandle = rclcpp_action::ServerGoalHandle<CountdownPark>;

/**
 * @brief 智能车整合控制节点
 * 
 * 功能：整合话题发布、服务提供、动作处理
 * 1. 发布车速指令到/cmd_vel
 * 2. 提供紧急停止服务
 * 3. 处理倒计时停车动作
 * 4. 监控电池状态
 */
class IntegratedControlNode : public rclcpp::Node {
public:
    IntegratedControlNode() : Node("integrated_control") {
        // 1. 初始化状态
        current_speed_ = 0.5f;
        battery_voltage_ = 12.8f;
        is_emergency_stop_ = false;
        
        // 2. 初始化话题发布者
        speed_publisher_ = this->create_publisher<geometry_msgs::msg::Twist>(
            "/cmd_vel", 10);
        
        battery_publisher_ = this->create_publisher<std_msgs::msg::Float32>(
            "/battery_voltage", 10);
        
        // 3. 初始化服务
        emergency_stop_service_ = this->create_service<EmergencyStop>(
            "/emergency_stop",
            std::bind(&IntegratedControlNode::handle_emergency_stop, this,
                     std::placeholders::_1, std::placeholders::_2));
        
        // 4. 初始化动作服务器
        action_server_ = rclcpp_action::create_server<CountdownPark>(
            this,
            "/countdown_park",
            std::bind(&IntegratedControlNode::handle_goal, this,
                     std::placeholders::_1, std::placeholders::_2),
            std::bind(&IntegratedControlNode::handle_cancel, this,
                     std::placeholders::_1),
            std::bind(&IntegratedControlNode::handle_accepted, this,
                     std::placeholders::_1));
        
        // 5. 创建定时器
        control_timer_ = this->create_wall_timer(
            100ms,
            std::bind(&IntegratedControlNode::control_loop, this)
        );
        
        battery_timer_ = this->create_wall_timer(
            1s,
            std::bind(&IntegratedControlNode::publish_battery_status, this)
        );
        
        RCLCPP_INFO(this->get_logger(), "智能车整合控制节点已启动！");
        RCLCPP_INFO(this->get_logger(), "当前速度：%.2f m/s", current_speed_);
        RCLCPP_INFO(this->get_logger(), "电池电压：%.2f V", battery_voltage_);
    }

private:
    // 状态变量
    float current_speed_;
    float battery_voltage_;
    bool is_emergency_stop_;
    
    // ROS2组件
    rclcpp::Publisher<geometry_msgs::msg::Twist>::SharedPtr speed_publisher_;
    rclcpp::Publisher<std_msgs::msg::Float32>::SharedPtr battery_publisher_;
    rclcpp::Service<EmergencyStop>::SharedPtr emergency_stop_service_;
    rclcpp_action::Server<CountdownPark>::SharedPtr action_server_;
    rclcpp::TimerBase::SharedPtr control_timer_;
    rclcpp::TimerBase::SharedPtr battery_timer_;
    
    // 控制循环
    void control_loop() {
        if (is_emergency_stop_) {
            // 紧急停止状态，发布零速度
            publish_zero_speed();
            return;
        }
        
        // 正常控制
        auto msg = geometry_msgs::msg::Twist();
        msg.linear.x = current_speed_;
        msg.angular.z = 0.1f;  // 轻微转弯
        
        speed_publisher_->publish(msg);
        
        static int count = 0;
        if (count++ % 10 == 0) {  // 每10次打印一次
            RCLCPP_DEBUG(this->get_logger(), 
                        "发布控制指令：速度=%.2f m/s", 
                        current_speed_);
        }
    }
    
    // 发布电池状态
    void publish_battery_status() {
        auto msg = std_msgs::msg::Float32();
        
        // 模拟电池电压缓慢下降
        battery_voltage_ -= 0.001f;
        if (battery_voltage_ < 11.0f) {
            battery_voltage_ = 12.8f;  // 重置，模拟充电
        }
        
        msg.data = battery_voltage_;
        battery_publisher_->publish(msg);
        
        RCLCPP_DEBUG(this->get_logger(), 
                    "发布电池电压：%.2f V", 
                    battery_voltage_);
    }
    
    // 发布零速度（紧急停止）
    void publish_zero_speed() {
        auto msg = geometry_msgs::msg::Twist();
        msg.linear.x = 0.0f;
        msg.angular.z = 0.0f;
        speed_publisher_->publish(msg);
    }
    
    // 紧急停止服务处理
    void handle_emergency_stop(
        const EmergencyStop::Request::SharedPtr request,
        EmergencyStop::Response::SharedPtr response) {
        
        RCLCPP_WARN(this->get_logger(), "收到紧急停止请求");
        
        if (request->stop_request) {
            is_emergency_stop_ = true;
            float stopped_speed = current_speed_;
            current_speed_ = 0.0f;
            
            response->success = true;
            response->message = "智能车已紧急停止";
            response->current_speed = stopped_speed;
            
            RCLCPP_ERROR(this->get_logger(), 
                        "紧急停止激活！停止前速度：%.2f m/s", 
                        stopped_speed);
            
            // 5秒后自动恢复
            this->create_wall_timer(
                5s,
                [this]() {
                    is_emergency_stop_ = false;
                    current_speed_ = 0.5f;
                    RCLCPP_INFO(this->get_logger(), "紧急停止解除，恢复正常控制");
                }
            );
        } else {
            response->success = false;
            response->message = "无效的停止请求";
            response->current_speed = current_speed_;
        }
    }
    
    // 动作服务器回调函数
    rclcpp_action::GoalResponse handle_goal(
        const rclcpp_action::GoalUUID & uuid,
        std::shared_ptr<const CountdownPark::Goal> goal) {
        
        RCLCPP_INFO(this->get_logger(), 
                   "收到倒计时停车目标：%d秒", 
                   goal->seconds);
        (void)uuid;
        
        if (is_emergency_stop_) {
            RCLCPP_WARN(this->get_logger(), 
                       "紧急停止状态中，拒绝新目标");
            return rclcpp_action::GoalResponse::REJECT;
        }
        
        return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
    }
    
    rclcpp_action::CancelResponse handle_cancel(
        const std::shared_ptr<GoalHandle> goal_handle) {
        RCLCPP_INFO(this->get_logger(), "收到取消请求");
        (void)goal_handle;
        return rclcpp_action::CancelResponse::ACCEPT;
    }
    
    void handle_accepted(const std::shared_ptr<GoalHandle> goal_handle) {
        std::thread{std::bind(&IntegratedControlNode::execute_countdown, this, goal_handle)}.detach();
    }
    
    void execute_countdown(const std::shared_ptr<GoalHandle> goal_handle) {
        auto feedback = std::make_shared<CountdownPark::Feedback>();
        auto result = std::make_shared<CountdownPark::Result>();
        
        auto goal = goal_handle->get_goal();
        int remaining_seconds = goal->seconds;
        
        for (int i = remaining_seconds; i >= 0; --i) {
            if (goal_handle->is_canceling()) {
                result->success = false;
                result->message = "倒计时停车被取消";
                result->final_speed = current_speed_;
                goal_handle->canceled(result);
                return;
            }
            
            // 更新反馈
            feedback->remaining_seconds = i;
            feedback->current_speed = current_speed_;
            goal_handle->publish_feedback(feedback);
            
            // 减速逻辑
            if (i <= 3) {
                current_speed_ = 0.5f * (i / 3.0f);
            }
            
            std::this_thread::sleep_for(1s);
        }
        
        // 完成停车
        current_speed_ = 0.0f;
        result->success = true;
        result->message = "倒计时完成，已安全停车";
        result->final_speed = current_speed_;
        goal_handle->succeed(result);
    }
};

int main(int argc, char *argv[]) {
    rclcpp::init(argc, argv);
    rclcpp::spin(std::make_shared<IntegratedControlNode>());
    rclcpp::shutdown();
    return 0;
}
```

## 3. Launch文件配置

### `launch/smart_car.launch.py`
```python
from launch import LaunchDescription
from launch_ros.actions import Node

def generate_launch_description():
    return LaunchDescription([
        # 整合控制节点
        Node(
            package='smart_car_core_demo',
            executable='integrated_control_node',
            name='integrated_control',
            output='screen',
            parameters=[
                {'control_frequency': 10.0},  # 10Hz控制频率
                {'max_speed': 1.0},           # 最大速度1.
